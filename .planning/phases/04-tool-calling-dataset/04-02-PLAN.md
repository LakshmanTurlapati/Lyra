files = sorted(glob.glob('datasets/tool-calling/cli-batch-*.jsonl'))

total_lines = 0
has_bash = False
has_git = False
has_file_ops = False
for f in files:
    result = subprocess.run([sys.executable, '-m', 'scripts.validate_format', f], capture_output=True, text=True)
    assert result.returncode == 0, f'Validation failed for {f}: {result.stderr}'
    with open(f) as fh:
        for line in fh:
            total_lines += 1
            conv = json.loads(line)
            for msg in conv['messages']:
                if msg.get('tool_calls'):
                    for tc in msg['tool_calls']:
                        cmd = tc['function'].get('arguments', {}).get('command', '')
                        if 'git' in cmd: has_git = True
                        elif any(c in cmd for c in ['ls', 'find', 'grep', 'curl', 'ps', 'top']): has_bash = True
                        elif any(c in cmd for c in ['cat', 'head', 'tail', 'mkdir', 'cp', 'mv']): has_file_ops = True
assert 800 <= total_lines <= 850, f'Total samples {total_lines} not in 800-850 range'

assert has_git, 'No git command samples found'
assert has_file_ops, 'No file operation samples found'
